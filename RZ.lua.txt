-- الصفحة الرئيسية
function HOME()
    HM1 = gg.choice({
        'BYPASS',
        'ESP',
        'MEMORY',
        '🔙 خروج Exit',
    }, nil, 'RZ SAFE')

    if HM1 == nil then return end
    if HM1 == 1 then antiban() end
    if HM1 == 2 then list1() end
    if HM1 == 3 then list2() end
    if HM1 == 4 then exit() end
end
